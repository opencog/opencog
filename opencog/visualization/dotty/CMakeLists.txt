	DottyModule.cc
)

	${ODBC_INCLUDE_DIRS}
)

	server
	${ATOMSPACE_LIBRARIES}

)

    INSTALL (TARGETS dotty DESTINATION "lib${LIB_DIR_SUFFIX}/opencog")

    INSTALL (TARGETS dotty DESTINATION "lib${LIB_DIR_SUFFIX}/opencog")
ELSE (WIN32)
    INSTALL (TARGETS dotty LIBRARY DESTINATION "lib${LIB_DIR_SUFFIX}/opencog")
ENDIF (WIN32)
